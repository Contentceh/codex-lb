async function checkSession() {
  const res = await fetch('/api/auth/session');
  const session = await res.json();
  
  if (Object.keys(session).length === 0) {
    console.warn("Сессия пуста. Возможные причины: вы не вошли в аккаунт или кука HttpOnly не была отправлена.");
  } else {
    console.log("Данные найдены:", session);
  }
}

checkSession();
